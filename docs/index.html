    let self = this;
    self.loading.full = true;
    self.loading.globalDisable = true;

        self.loading.full = false;
        self.loading.globalDisable = false;
    }, 5000);
}

    let idx = this.uploader.files.findIndex((x) => { return x.id == id; });            
    
    if (idx >= 0) {
        this.uploader.files.splice(idx, 1);
    }
},
